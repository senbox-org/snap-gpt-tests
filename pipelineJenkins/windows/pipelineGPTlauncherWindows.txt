/**
 * Launch jobs for every json file listed in jsonString separated by '\n'
 */
def launchJobs(jsonString, scope, outputDir, saveOutput, debug) {
	println saveOutput
    println "List of sequentials Json file : " + jsonString
    jsonString = jsonString.trim()
    jsonList = jsonString.split("\n")
    num = 0
    status = true
    sh script:"mkdir -p ${outputDir}/OUT", label: "create report directory"
    for (int i=0; i < jsonList.size(); i++) {
        item = jsonList[i]
        def currentJsonFile = "" + item
        if (currentJsonFile.trim() != "") {
			// Test commit message for flags
			try{
				echo """export LD_LIBRARY_PATH=. && ${PYTHON_CMD} -u "${outputDir}/pygpt/snap_gpt_test.py" '${JAVA_CMD}' '-Dncsa.hdf.hdflib.HDFLibrary.hdflib=${SNAP_INSTALL_ROOT_DIR2}/snap/modules/lib/amd64/jhdf.${LIB_EXTENSION} -Dncsa.hdf.hdf5lib.H5.hdf5lib=${SNAP_INSTALL_ROOT_DIR2}/snap/modules/lib/amd64/jhdf5.${LIB_EXTENSION} -cp ./gpt-tests-executer/target/TestOutput.jar' org.esa.snap.test.TestOutput \"${GPT_EXEC_PROPERTIES}\" \"${scope}\" \"${currentJsonFile}\" \"${outputDir}/report\" ${saveOutput}"""
				sh """export LD_LIBRARY_PATH=. && ${PYTHON_CMD} -u "${outputDir}/pygpt/snap_gpt_test.py" '${JAVA_CMD}' '-Dncsa.hdf.hdflib.HDFLibrary.hdflib=${SNAP_INSTALL_ROOT_DIR2}/snap/modules/lib/amd64/jhdf.${LIB_EXTENSION} -Dncsa.hdf.hdf5lib.H5.hdf5lib=${SNAP_INSTALL_ROOT_DIR2}/snap/modules/lib/amd64/jhdf5.${LIB_EXTENSION} -cp ./gpt-tests-executer/target/TestOutput.jar' org.esa.snap.test.TestOutput \"${GPT_EXEC_PROPERTIES}\" \"${scope}\" \"${currentJsonFile}\" \"${outputDir}/report\" ${saveOutput}"""
			} catch (all) {
                echo "A test from the testset \"${currentJsonFile}\" failed"
                status = false
            }
		}
        num++
    }
    if (!status) {
        throw new Exception("At least one test failed")
    }
}
